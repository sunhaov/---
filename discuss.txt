        function batchDel() {

            var topic = document.getElementsByClassName("checkTopic");

            var createrId = "";

            var topicIds = new Array();

            var createIds = new Array();

            for (var i = 0; i < topic.length; i++) {

                if (topic[i].checked) {

                    createrId = topic[i].id;

                    createrId = $("#" + createrId).attr("curCreaterId");

                    topicIds.push(topic[i].value);

                    createIds.push(createrId);

                }

            }

            if (createIds.length == 0 || topicIds.length == 0) {

                alert("未选中讨论");

                return;

            }

            if (confirm("确定删除吗？")) {

                jQuery.ajax({

                    type: "get",

                    url: "/bbscircle/batchDeleteTopic",

                    data: {

                        "clazzid": 9263900,

                        "topicIds": topicIds.toString(),

                        "createIds": createIds.toString(),

                        "courseId": 204686578

                    },

                    dataType: 'json',

                    success: function (data) {

                        if (data.result == 1) {

                            window.location.reload();

                        } else {

                            alert(data.msg);

                        }

                    }

                });

            }

        };
